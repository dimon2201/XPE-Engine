31 void* cMemoryPool::Allocate(const usize size)
32 {

35#endif
36
37 for (auto& alloc: m_Allocs)
38 {
39 if (alloc.FreeFlag == 1u && alloc.AllocByteWidth >= size)
40 {
41 alloc.FreeFlag = 255u;
42 alloc.OccupiedByteWidth = size;
43
44 return alloc.Address;
45 }
46

52 newAlloc.AllocByteWidth = alloc.AllocByteWidth - alloc.OccupiedByteWidth;
