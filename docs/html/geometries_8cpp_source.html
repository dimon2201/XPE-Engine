158 sSphere::sSphere(s32 xSegments, s32 ySegments)
159 {
160 PrimitiveTopology = ePrimitiveTopology::TRIANGLE_STRIP;
161
162 auto& vertices = Vertices;
163 vertices.resize((xSegments + 1) * (ySegments + 1));
164
165 int i = 0;
166 for (u32 x = 0; x <= xSegments; x++)
167 {
168 for (u32 y = 0; y <= ySegments; y++)
169 {
170 auto& V = vertices[i];
171
172 float xSegment = (float)x / (float)xSegments;
173 float ySegment = (float)y / (float)ySegments;
174
175 float xPos = cos(xSegment * 2.0f * K_PI) * sin(ySegment * K_PI);
176 float yPos = cos(ySegment * K_PI);
177 float zPos = sin(xSegment * 2.0f * K_PI) * sin(ySegment * K_PI);
178
179 V.Position = { xPos, yPos, zPos };
180 V.UV = { xSegment, ySegment };
181 V.Normal = glm::normalize(V.Position);
182
183 // calculate tangent
184 // todo: it will not always give a correct tangents.
185 // todo: we need to check for different case of [(theta,phi+pi/2);(theta+pi/2,phi);(theta+pi/2,phi+pi/2)]
186 glm::vec3 tangent;
187 float r = V.Position.length();
188 float theta = acos(zPos / r);
189 float phi = atan2(yPos, xPos) + K_PI * 0.5;
190 tangent.x = sin(theta) * cos(phi);
191 tangent.y = sin(theta) * sin(phi);
192 tangent.z = cos(theta);
193 V.Tangent = tangent;
194
195 i++;
196 }
197 }
198

200 vector<u32> indexList;
201 indexList.resize(xSegments * ySegments + 1);
202

204 {
