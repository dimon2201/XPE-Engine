48 void cTaskDispatcher::Dispatch(u32 tasksPerThread, u32 taskSize, const sTask& task)
49 {
50 if (taskSize == 0 || tasksPerThread == 0) {
51 return;
52 }
53
54 u32 taskGroups = (tasksPerThread + taskSize - 1) / taskSize;
55 m_TasksTodo += taskGroups;
56
57 for (u32 i = 0; i < taskGroups; ++i) {
58 // create task group
59 sTask taskGroup;
60 taskGroup.Todo = [i, task, taskSize, tasksPerThread]() {
61 u32 groupJobOffset = i * taskSize;
62 u32 groupJobEnd = std::min(groupJobOffset + taskSize, tasksPerThread);
63 for (u32 j = groupJobOffset; j < groupJobEnd; ++j) {
64 task.Todo();
65 }
66 };
67
68 // try to push new group as a task until its pushed
69 while (!m_TaskBuffer.Push(taskGroup)) {
70 Poll();
71 }
72
73 m_WakeCondition.notify_one();
74 }
75 }

100 void cTaskDispatcher::InitThread(u32 workerId, const char* name, cThread::ePriority priority)
101 {
102 std::thread worker([this]() {
103 sTask task;
104 while (true) {
105 if (m_TaskBuffer.Pop(task)) {
106 // do task and update worker state
107 task.Todo();
108 m_TasksDone.fetch_add(1);
