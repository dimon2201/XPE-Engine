26 PxFilterObjectAttributes attributes1, PxFilterData filterData1,
27 PxPairFlags& pairFlags, const void* constantBlock, PxU32 constantBlockSize)
28 {
29 // let triggers through
30 if (PxFilterObjectIsTrigger(attributes0) || PxFilterObjectIsTrigger(attributes1))
