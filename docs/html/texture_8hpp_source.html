208 sAtlas::sLocation sAtlas::AddCell(Args &&... args)
209 {
210 sCell cell = sCell(std::forward<Args>(args)...);
211
212 if (cell.LayerIndex >= Layers.size()) {
213 LogWarning("Atlas with index={} is not initialized. Initialize atlasLayer before adding this cell!", cell.LayerIndex);
214 return {};
215 }
216
217 if (cell.Texture == nullptr || cell.Texture->Pixels == nullptr) {
218 LogWarning("Cell texture or texture pixels are not initialized. Initialize cell texture before adding this cell!");
219 return {};
220 }
221
222 sTextureLayer& atlasLayer = Layers[cell.LayerIndex];
223 sTextureLayer cellTexture = cell.Texture->Clone();
224 cellTexture.Resize(Format, cell.Size.x, cell.Size.y);
225
226 s32 cellX = cell.Position.x;
227 s32 cellY = cell.Position.y;
228 s32 cellWidth = cell.Size.x;
229 s32 cellHeight = cell.Size.y;
230 u8* atlasPixels = static_cast<u8*>(atlasLayer.Pixels);
231 u8* cellPixels = static_cast<u8*>(cellTexture.Pixels);
232 Channels = k_ChannelTable.at(Format);
233 int bpc = k_BppTable.at(Format) / Channels; // bytes per channel
234
235 for (int y = 0; y < cellHeight; ++y) {
236 for (int x = 0; x < cellWidth; ++x) {
237 int cellId = (y * cellWidth + x) * Channels;
238 int atlasX = x + cellX;
239 int atlasY = y + cellY;
240 int atlasId = (atlasY * atlasLayer.Width + atlasX) * Channels;
241 for (int c = 0; c < Channels; ++c) {
242 atlasPixels[atlasId + c * bpc] = cellPixels[cellId + c * bpc];
243 }
244 }
245 }
246
247 cellTexture.Free();
248
249 sLocation location;
250 location.LayerIndex = cell.LayerIndex;
251 location.UV[0] = glm::fvec2(cellX, cellY) / glm::fvec2(atlasLayer.Width, atlasLayer.Height);
252 location.UV[1] = glm::fvec2(cellX, cellY + cellHeight) / glm::fvec2(atlasLayer.Width, atlasLayer.Height);
253 location.UV[2] = glm::fvec2(cellX + cellWidth, cellY + cellHeight) / glm::fvec2(atlasLayer.Width, atlasLayer.Height);
254 location.UV[3] = glm::fvec2(cellX + cellWidth, cellY) / glm::fvec2(atlasLayer.Width, atlasLayer.Height);
255 return location;
256 }
