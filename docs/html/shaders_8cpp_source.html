18 void cInstancingShader::DrawInstanced(
19 cScene* scene,
20 const sGeometryInfo& geometryInfo,
21 const std::function<void(EntityID entityId, sRenderInstance&)>& callback
22 ) {
23 usize entityCount = geometryInfo.Entities.size();
24 usize instanceCount = 0;
25 m_InstanceBuffer.Clear();
26 for (usize i = 0 ; i < entityCount ; i++)
27 {
28 auto& entity = geometryInfo.Entities[i];
29
30 if (!scene->HasAnyComponent<CVisible>(entity))
31 continue;
32
33 sRenderInstance instance;
34 instance.ModelMatrix = cMathManager::UpdateModelMatrix(scene->GetComponent<CTransform>(entity));
35 instance.NormalMatrix = cMathManager::UpdateNormalMatrix(instance.ModelMatrix);
36 callback(entity, instance);
37
38 m_InstanceBuffer.Add(instance);
39
40 instanceCount++;
41 }
42 m_InstanceBuffer.Flush();
43
44 context::BindPrimitiveTopology(geometryInfo.PrimitiveTopology);
45 context::VSBindBuffer(m_InstanceBuffer);
46
47 if (geometryInfo.IndexCount == 0) {
48 context::DrawVertexed(geometryInfo.VertexCount, instanceCount, geometryInfo.VertexOffset);
49 } else {
50 context::DrawIndexed(geometryInfo.IndexCount, instanceCount, geometryInfo.VertexOffset, geometryInfo.IndexOffset);
51 }
52
53 context::VSUnbindBuffer(m_InstanceBuffer);
54 }

576 bool enableTexture,
577 const sAtlas::sLocation& atlasLocation,

582 ) {

704 bool cWidgetShader::IsHovered2D(const glm::mat4 &modelMatrix, const glm::vec2 &mousePos)
705 {
706 glm::vec4 v1 = modelMatrix * glm::vec4(m_Quad.Vertices[0].Position, 1.0);
707 glm::vec4 v2 = modelMatrix * glm::vec4(m_Quad.Vertices[1].Position, 1.0);
708 glm::vec4 v3 = modelMatrix * glm::vec4(m_Quad.Vertices[2].Position, 1.0);

834 void cTextShader::UpdateText2D(
835 glm::mat4 projection,
836 sTransform transform,
837 u32 textIndex,
838 sFont& font,
839 const string& textString,
840 const glm::vec4& textColor,
841 bool fillFrame
842 ) {
843 if (textString.empty()) return;
844
845 usize charsCount = textString.size();
846 const char* chars = textString.c_str();
847 glm::mat4 modelMatrix;
848 sText text;
849 text.Projection = projection;
850 text.Color = textColor;
851
852 float frameWidth = (float) cWindowManager::GetFrameWidth();
853 float frameHeight = (float) cWindowManager::GetFrameHeight();
854
855 transform.Position = {
856 transform.Position.x * frameWidth,
857 transform.Position.y * frameHeight,
858 transform.Position.z
859 };
860
861 if (fillFrame) {
862 transform.Scale = {
863 transform.Scale.x * frameWidth,
864 transform.Scale.y * frameHeight,
865 transform.Scale.z
866 };
867 }
868
869 text.ModelMatrix = cMathManager::UpdateModelMatrix(transform);
870
871 m_TextBuffer.Add(text);
872
873 glm::vec2 advance = { 0, 0 };
874
875 for (usize i = 0; i < charsCount; i++)
876 {
877 char c = chars[i];
878
879 // find glyph by char
880 auto it = font.AlphaBet.find(c);
881 if (it == font.AlphaBet.end()) {
882 continue;
883 }
884 sFont::sGlyph glyph = it->second;
885
886 xpe::render::sChar character;
887 character.TextIndex = textIndex;
888 character.GlyphSize = font.GlyphSize;
889 character.Width = glyph.Width;
890 character.Height = glyph.Height;
891 character.Left = glyph.Left;
892 character.Top = glyph.Top;
893 character.AdvanceX = advance.x;
894 character.AdvanceY = advance.y;
895 character.AtlasXOffset = glyph.AtlasXOffset;
896 character.AtlasYOffset = glyph.AtlasYOffset;
897
898 // Tab
899 if (c == '\t')
900 {
901 advance.x += font.WhitespaceOffset * transform.Scale.x;
902 }
903
904 // New line
905 else if (c == '\n')
906 {
907 advance.x = 0.0f;
908 advance.y += (font.MaxGlyphHeight + font.NewLineOffset) * transform.Scale.y;
909 }
910
911 else
912 {
913 advance.x += (glyph.AdvanceX / 64.0f) * transform.Scale.x;
914 m_CharBuffer.Add(character);
915 }
916 }
917 }
