9 string cFileManager::ReadFileWithIncludes(const string& path, string includeIdentifier)
10 {
11 includeIdentifier += ' ';
12 static bool recursive = false;
13 string fullSrc = "";
14 std::ifstream file;
15
16 file.open(path.c_str());
17
18 if (!file.is_open()) {
19 LogError("Failed to open file {}", path.c_str());
20 return fullSrc;
21 }
22
23 string lineBuffer;
24 while (std::getline(file, lineBuffer)) {
25 // Look for the new mShader include identifier
26 if (lineBuffer.find(includeIdentifier) != lineBuffer.npos) {
27 // Remove the include identifier, this will cause the path to remain
28 lineBuffer.erase(0, includeIdentifier.size());
29
30 // The include path is relative to the current mShader file path
31 string pathOfThisFile;
32 GetFilepath(path, pathOfThisFile);
33 lineBuffer.insert(0, pathOfThisFile);
34
35 // By using recursion, the new include file can be extracted
36 // and inserted at this location in the mShader source code
37 recursive = true;
38 fullSrc += ReadFileWithIncludes(lineBuffer, includeIdentifier) + '\n';
39
40 // Do not add this line to the mShader source code, as the include
41 // path would generate a compilation issue in the final source code
42 continue;
43 }
44
45 fullSrc += lineBuffer + '\n';
46 }
47
48 // Only add the null terminator at the end of the complete file,
49 // essentially skipping recursive function calls this way
50 if (!recursive)
51 fullSrc += '\0';
52
53 file.close();
54
55 return fullSrc;
56 }
